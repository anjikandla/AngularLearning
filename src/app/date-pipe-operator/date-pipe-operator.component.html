Date using Pipe Operator

  • {{date|date:'short'}} //short
  • {{date|date:'shortDate'}} //shortDate
  • {{date|date:'shortTime'}} //shortTime
  • {{date|date:'medium'}} //medium
  • {{date|date:'mediumDate'}} //mediumDate
  • {{date|date:'mediumTime'}} //mediumTime
  • {{date|date:'long'}} //long
  • {{date|date:'longDate'}} //longDate
  • {{date|date:'longTime'}} //longTime
